9239: Egy "Windows"-os megoldas (Windows Scripting Host JScript). Az alabbi programot elmentve, ra drag-and-drop-olhatok a bemeneti file-ok. A kimenetet kulon file-okba irja.
// script.js
var fso = new ActiveXObject("Scripting.FileSystemObject");
var RD = 1;
var WR = 2;
function process(name)
{
if(name.match(/\./)==null)
outname = name + " eredmeny";
else
outname = name.replace(/\.([^.]+)$/, " eredmeny.$1");
fi = fso.OpenTextFile(name, RD);
fo = fso.OpenTextFile(outname, WR, true);
while(!fi.AtEndOfStream)
fo.WriteLine(fi.ReadLine().replace(/(^.)|(...$)/g, ""));
fi.Close();
fo.Close();
}
var args = WScript.Arguments;
for(i = 0; i < args.length; i++)
process(args(i));
