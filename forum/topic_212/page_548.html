while read ; do
for www in $REPLY ; do
w=`echo $www | sed 's/[^a-zA-Z]*$//g'`
i=0
while [ $i -lt ${#w} ]; do
ww[$i]=${w:$i:1}
i=$((i+1))
done
if [ ${#w} -gt 3 ] ; then
i=0
while [ $i -lt ${#w} ]; do
s1=$(($(($RANDOM%$((${#w}-2))))+1))
s2=$(($(($RANDOM%$((${#w}-2))))+1))
t=${ww[$s1]}
ww[$s1]=${ww[$s2]}
ww[$s2]=$t
i=$((i+1))
done
fi
i=0
while [ $i -lt ${#w} ]; do
echo -n ${ww[$i]}
i=$((i+1))
done
echo -n ${www#$w}
echo -n " "
done
echo
done
